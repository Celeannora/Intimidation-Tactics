```python
import sqlite3
import duckdb
import json
from typing import List, Dict, Optional

class MagicCardDB:
    def __init__(self, db_path: str = "mtg_cards.db"):

        self.duck = duckdb.connect(":memory:")
        self._init_schema()
    
    def _init_schema(self):
        self.conn.executescript("""
            CREATE TABLE IF NOT EXISTS cards (
                id TEXT PRIMARY KEY,
                name TEXT UNIQUE NOT NULL,
                mana_cost TEXT,
                cmc REAL,
                type_line TEXT,
                oracle_text TEXT,
                colors TEXT,
                color_identity TEXT,
                power TEXT,
                toughness TEXT,
                loyalty TEXT,
                keywords TEXT,
                set_name TEXT,
                rarity TEXT,
                prices_usd REAL,
                prices_usd_foil REAL,
                prices_eur REAL,
                edhrec_rank INTEGER,
                legality_standard TEXT,
                legality_commander TEXT,
                legality_modern TEXT,
                legality_pauper TEXT,
                last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            );
            
            CREATE TABLE IF NOT EXISTS banlist (
                format TEXT,
                card_name TEXT,
                banned_since TEXT,
                PRIMARY KEY (format, card_name)
            );
            
            CREATE TABLE IF NOT EXISTS synergies (
                card_a TEXT,
                card_b TEXT,
                synergy_type TEXT,
                confidence REAL,
                UNIQUE(card_a, card_b)
            );
            
            CREATE INDEX idx_color_identity ON cards(color_identity);
            CREATE INDEX idx_cmc ON cards(cmc);
            CREATE INDEX idx_type ON cards(type_line);
```

```python
# engine/deck_rules.py
from dataclasses import dataclass
from typing import List, Set, Tuple
import re

@dataclass
class ValidationResult:
    is_valid: bool
    errors: List[str]
    warnings: List[str]
    manacurve: Dict[int, int]  # cmc -> count
    color_pie: Dict[str, int]  # color -> count
    mana_sources: Dict[str, int]  # color -> number of sources

class DeckRules:
    FORMAT_RULES = {
        "commander": {
            "min_size": 99,
            "max_size": 99,
            "max_copies": 1,
            "needs_commander": True,
            "color_identity_strict": True,
            "sideboard_size": 0
        },
        "standard": {
            "min_size": 60,
            "max_size": None,
            "max_copies": 4,
            "color_identity_strict": False,
            "sideboard_size": 15
        }
    }
    
    def validate_deck(self, cards: List[Dict], format: str, commander: str = None) -> ValidationResult:
        errors = []
        warnings = []
        
        # Edge Case: Companion check
        if self._has_companion(cards):
            companion = self._get_companion(cards)
            companion_condition = self._extract_companion_condition(companion)
            if not self._check_companion_condition(cards, companion_condition):
                errors.append(f"Companion {companion['name']} condition not met")
        
        # Edge Case: MDFC handling
        mdfc_bonus = self._count_mdfc_backfaces(cards)
        
        # Edge Case: Token generators don't count as deck slots
        token_generators = [c for c in cards if self._is_token_generator(c)]
        
        # Color identity from commander (for Commander format)
        if commander:
            commander_colors = self._parse_color_identity(commander)
            for card in cards:
                card_colors = self._parse_color_identity(card)
                if not card_colors.issubset(commander_colors):
                    # Edge Case: MDFC front face only counts for color identity
                    if not self._is_mdfc_front(card):
                        errors.append(f"Color identity violation: {card['name']} ({card_colors}) not in commander's ({commander_colors})")
                    else:
                        # Allow MDFC if front face is legal
                        front_colors = self._get_mdfc_front_colors(card)
                        if not front_colors.issubset(commander_colors):
                            errors.append(...)
        
        return ValidationResult(
            is_valid=len(errors) == 0,
            errors=errors,
            warnings=warnings,
            manacurve=self._compute_curve(cards),
            color_pie=self._compute_color_pie(cards),
            mana_sources=self._compute_mana_sources(cards)
        )
    
    def _compute_mana_sources(self, cards: List[Dict]) -> Dict[str, int]:
        """Calculate color sources including fixing lands and mana rocks"""
        sources = {"W": 0, "U": 0, "B": 0, "R": 0, "G": 0, "C": 0, "multi": 0}
        for card in cards:
            if "Land" in card["type_line"]:
                colors_produced = self._get_land_colors(card)
                # Edge Case: Lands that produce multiple colors count fractionally
                for color in colors_produced:
                    sources[color] += 1.0 / len(colors_produced)
            if "Artifact" in card["type_line"] or "Enchantment" in card["type_line"]:
                if self._is_mana_source(card):
                    sources[card.get("colors", "")[0]] += 1 if len(card.get("colors", "")) == 1 else 0.5
        return {k: int(v) for k, v in sources.items()}
```

3. "How would you implement a pipeline that: batches card pairs for synergy analysis (up to 100 pairs per LLM call to reduce overhead), caches results in the `synergies` table, and falls back to keyword-based heuristic scoring when LLM is unavailable?"

4. "What prompt engineering techniques prevent hallucination of non-existent Magic cards? Include: giving LLM a truncated card list from the database, instructing it to only use cards from the provided data, and post-processing regex validation against known card names."

```python
# llm/local_llm.py
from llama_cpp import Llama
from llama_cpp.llama_grammar import LlamaGrammar
import json
import re
from typing import List, Dict, Optional
import threading
import queue

class OfflineMagicLLM:
    def __init__(self, model_path: str = "models/mistral-7b-instruct.Q4_K_M.gguf"):
        self.model = Llama(
            model_path=model_path,
            n_ctx=4096,
            n_threads=8,
            n_gpu_layers=35,  # Offload to GPU if available
            verbose=False,
            use_mmap=True
        )
        self.response_queue = queue.Queue()
        self._load_synergy_grammar()
    
    def _load_synergy_grammar(self):
        """GBNF grammar to force structured output and prevent hallucinations"""
        self.grammar = LlamaGrammar.from_string("""
            root ::= "{" ws "\"synergies\"" ws ":" ws "[" ws synergy_pair ("," ws synergy_pair)* ws "]" ws "}"
            synergy_pair ::= "{" ws "\"card_a\"" ws ":" ws string ws "," ws "\"card_b\"" ws ":" ws string ws "," ws "\"score\"" ws ":" ws float ws "," ws "\"reason\"" ws ":" ws string ws "}"
            string ::= "\"" [^\"]* "\""
            float ::= [0-9] "." [0-9]+
            ws ::= [ \t\n\r]*
        """)
    
    def analyze_synergy_batch(self, card_pairs: List[tuple], batch_size: int = 100) -> List[Dict]:
        """Batch synergy analysis with fallback"""
        results = []
        for i in range(0, len(card_pairs), batch_size):
            batch = card_pairs[i:i+batch_size]
            
            # Check cache first
            cached = self._check_cache(batch)
            results.extend([r for r in cached if r is not None])
            
            uncached = [(a,b) for (a,b) in batch if (a,b) not in {r['card_a'], r['card_b']} for r in cached]
            
            if uncached:
                try:
                    llm_results = self._llm_analyze(uncached)
                    self._update_cache(llm_results)
                    results.extend(llm_results)
                except Exception:
                    # Fallback to heuristic scoring
                    heuristic = self._heuristic_analyze(uncached)
                    results.extend(heuristic)
        
        return results
    
    def _llm_analyze(self, pairs: List[tuple]) -> List[Dict]:
        """Constrained generation for synergy analysis"""
        prompt = self._build_synergy_prompt(pairs)
        
        # Stream response to avoid network timeout
        response = ""
        for token in self.model(prompt, grammar=self.grammar, stream=True):
            response += token["choices"][0]["text"]
            if len(response) > 4000:  # Safety limit
                break
        
        try:
            parsed = json.loads(response)
            # Post-processing: validate card names exist in our database
            validated = []
            for item in parsed["synergies"]:
                if self._card_exists(item["card_a"]) and self._card_exists(item["card_b"]):
                    validated.append({
                        "card_a": item["card_a"],
                        "card_b": item["card_b"],
                        "score": max(0.0, min(1.0, float(item["score"]))),  # Clamp
                        "reason": item.get("reason", "")
                    })
                else:
                    # Hallucination detected - replace with heuristic
                    validated.append(self._heuristic_for_pair(item["card_a"], item["card_b"]))
            return validated
        except (json.JSONDecodeError, KeyError):
            return self._heuristic_analyze(pairs)
    
    def _build_synergy_prompt(self, pairs: List[tuple]) -> str:
        """Prompt with guardrails against hallucination"""
        card_names = set()
        for a, b in pairs:
            card_names.add(a)
            card_names.add(b)
        
        # Only include cards that exist in our database
        valid_cards = self._get_known_cards(list(card_names))
        
        return f"""<s>[INST] You are a Magic: The Gathering deck synergy analyzer. You ONLY analyze cards from this exact list (do not invent any cards):

Known cards: {json.dumps(valid_cards)}

Analyze these card pairs for synergy (0.0 = no synergy, 1.0 = perfect synergy). Consider:
- Combo potential (infinite combos, two-card combos)
- Tribal synergy (shares creature type)
- Mana curve interaction
- Color fixing needs
- Theme alignment (graveyard, artifacts, spellslinger, etc.)

Pairs to analyze: {json.dumps(pairs)}

Respond ONLY with valid JSON matching this exact schema (no other text):
{{"synergies": [{{"card_a": string, "card_b": string, "score": float, "reason": string}}, ...]}}

IMPORTANT: If a card name is NOT in the provided list, use 'UNKNOWN_CARD' instead and assign score 0.0.
[/INST]"""
```

```python
                    status += f"<ul>{warnings}</ul>"
            
            return len(self.current_deck), status, fig_curve, fig_pie
        
        # Chain updates: any interaction that modifies deck triggers validation
        for component in [components['format_dropdown'], components['commander_input']]:
            component.change(
                update_validation,
                inputs=[components['format_dropdown'], components['commander_input']],
                outputs=[components['deck_size'], components['validation_status'], 
                         components['mana_curve_plot'], components['color_pie']]
            )
        
        # Smart Builder generation with streaming
        async def generate_decks(format, commander, budget, playstyle, theme):
            components['loading_indicator'].visible = True
            yield components['loading_indicator']
            
            # Stream generation progress
            def generator_thread(q):
                try:
                    decks = self.llm.generate_deck_lists(
                        format=format,
                        commander=commander,
                        budget=budget,
                        playstyle=playstyle,
                        theme=theme,
                        num_options=3
                    )
                    q.put(decks)
                except Exception as e:
                    q.put(f"ERROR: {str(e)}")
            
            result_queue = queue.Queue()
            thread = threading.Thread(target=generator_thread, args=(result_queue,))
            thread.start()
            
            while thread.is_alive():
                yield components['loading_indicator']  # Could update with progress percentage
                await asyncio.sleep(0.5)
            
            decks = result_queue.get()
            components['loading_indicator'].visible = False
            yield components['deck_options'], components['deck_preview']
        
        generate_btn.click(
            generate_decks,
            inputs=[components['format_dropdown'], components['commander_input'],
                    components['budget_slider'], components['playstyle'], components['deck_theme']],
            outputs=[components['deck_options'], components['deck_preview']]
        )
```

```python
# performance/cache_manager.py
import sqlite3
import hashlib
import json
from collections import OrderedDict
from typing import Any, Optional
import threading
import time

class SynergyCache:
    def __init__(self, db_path: str = "cache/synergies.db"):
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS synergy_cache (
                card_a TEXT,
                card_b TEXT,
                score REAL,
                reason TEXT,
                cached_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                PRIMARY KEY (card_a, card_b)
            )
        """)
        self.conn.execute("PRAGMA journal_mode=WAL")
        self.lock = threading.Lock()
        
        # In-memory LRU for hot cache
        self.memory_cache = OrderedDict()
        self.memory_cache_size = 10000
        self.memory_cache_ttl = 300  # 5 minutes
    
    def get_synergy(self, card_a: str, card_b: str) -> Optional[dict]:
        # Check memory cache first
        key = tuple(sorted([card_a, card_b]))
        if key in self.memory_cache:
            entry = self.memory_cache[key]
            if time.time() - entry['time'] < self.memory_cache_ttl:
                # Move to end (most recently used)
                self.memory_cache.move_to_end(key)
                return entry['data']
            else:
                del self.memory_cache[key]
        
        # Check disk cache
        with self.lock:
            cursor = self.conn.execute(
                "SELECT score, reason FROM synergy_cache WHERE card_a=? AND card_b=?",
                (card_a, card_b)
            )
            row = cursor.fetchone()
            if row:
                data = {"card_a": card_a, "card_b": card_b, "score": row[0], "reason": row[1]}
                # Add to memory cache
                self._add_to_memory(key, data)
                return data
        return None
    
    def cache_synergy_batch(self, results: list):
        with self.lock:
            self.conn.executemany(
                """INSERT OR REPLACE INTO synergy_cache 
                   (card_a, card_b, score, reason) VALUES (?, ?, ?, ?)""",
                [(r['card_a'], r['card_b'], r['score'], r['reason']) for r in results]
            )
            self.conn.commit()
        
        # Update memory cache
        for r in results:
            key = tuple(sorted([r['card_a'], r['card_b']]))
            self._add_to_memory(key, r)
    
    def _add_to_memory(self, key: tuple, data: dict):
        if len(self.memory_cache) >= self.memory_cache_size:
            # Evict oldest
            self.memory_cache.popitem(last=False)
        self.memory_cache[key] = {'data': data, 'time': time.time()}
```

## Phase 3 Implementation: Offline-First LLM Integration

This phase focuses on the core LLM module – `llm/local_llm.py` – plus supporting files for prompt engineering, response parsing, synergy caching, and fallback heuristics.  
All code assumes the `engine/` modules (card database, deck rules) are already in place.

### 1. Critical Answers to Phase 3 Questions

**Q1: Managing GGUF model with llama-cpp-python (CPU/GPU, caching, streaming)**  
- Use `llama_cpp.Llama` with `n_ctx=4096` for phi‑3‑mini / `n_ctx=8192` for mistral.  
- Offload GPU layers via `n_gpu_layers` (e.g., 35 for an 8‑GB card).  
- Enable `use_mmap=True` for fast memory mapping.  
- Implement a thread‑safe **response queue** and yield tokens one‑by‑one to the UI (non‑blocking).  
- Cache LLM responses in an **LRU cache** with disk persistence (SQLite) – see `cache_manager.py`.  
- For streaming, use `model(prompt, stream=True)` and yield tokens; the UI thread consumes them.

**Q2: Prompt system with structured JSON (GBNF grammar)**  
- Define a **GBNF grammar** that forces the LLM to output exactly:  
  `{"synergies": [{"card_a": str, "card_b": str, "score": float, "reason": str}, ...]}`  
- The grammar rejects free text, prevents hallucinated card names by only allowing strings that match known names (passed in the prompt).  
- Include **anti‑synergy** detection by instructing the model to assign scores < 0.3 for conflicting cards (e.g., two different tribal lords, or cards that counter each other).  
- If a card name is not in the provided list, the grammar can still accept it; we then post‑process and replace with `UNKNOWN_CARD`.

**Q3: Batching, caching, and fallback**  
- Batch up to 100 card pairs per LLM call (to fit in context).  
- Check **SQLite cache** first (disk + in‑memory LRU).  
- If LLM fails (timeout, error, corrupted output), fall back to **keyword‑based heuristic scorer** that checks:  
  - Shared creature types / subtypes  
  - Mana cost synergy (e.g., low‑CMC spells + high‑CMC payoffs)  
  - Explicit combo keywords (“whenever you cast”, “when this enters”, etc.)  
- Heuristic scoring is fast and runs locally without the LLM.

**Q4: Anti‑hallucination techniques**  
- **Truncated card list**: In the prompt we include only the names of cards from the current deck plus a small context set (no full database).  
- **Explicit instruction**: “You ONLY analyze cards from this exact list. Do not invent any cards.”  
- **Post‑processing regex**: After parsing JSON, we verify every `card_a` / `card_b` against a set of known names (loaded from the DB). Non‑matching names are replaced with `UNKNOWN_CARD` and score set to 0.0.  
- **Grammar constraints**: The GBNF grammar limits characters; names with special characters are still allowed, but the model cannot inject extra text.

```python
import json
import queue
import threading
import time
from typing import List, Dict, Optional, Tuple
from llama_cpp import Llama
from llama_cpp.llama_grammar import LlamaGrammar
from .response_parser import validate_card_names
from .prompt_templates import SynergyPromptBuilder
from ..performance.cache_manager import SynergyCache
from ..engine.card_db import MagicCardDB  # for known card names

class OfflineMagicLLM:
    def __init__(self,
                 model_path: str = "models/phi-3-mini-4k-instruct.Q4_K_M.gguf",
                 db_path: str = "mtg_cards.db",
                 cache_path: str = "cache/synergies.db",
                 n_gpu_layers: int = 0,    # set >0 for GPU offloading
                 n_threads: int = 8,
                 n_ctx: int = 4096):
        self.db = MagicCardDB(db_path)
        self.cache = SynergyCache(cache_path)
        self.prompt_builder = SynergyPromptBuilder()
        self.model = Llama(
            model_path=model_path,
            n_ctx=n_ctx,
            n_threads=n_threads,
            n_gpu_layers=n_gpu_layers,
            verbose=False,
            use_mmap=True
        )
        self.grammar = self._load_grammar()
        self.response_queue = queue.Queue()
        self.lock = threading.Lock()

    def _load_grammar(self) -> LlamaGrammar:
        """GBNF grammar for structured synergy output."""
        grammar_text = """
        root ::= "{" ws "\\"synergies\\"" ws ":" ws "[" ws synergy_pair ("," ws synergy_pair)* ws "]" ws "}"
        synergy_pair ::= "{" ws "\\"card_a\\"" ws ":" ws string ws "," ws "\\"card_b\\"" ws ":" ws string ws "," ws "\\"score\\"" ws ":" ws float ws "," ws "\\"reason\\"" ws ":" ws string ws "}"
        string ::= "\\"" [^\\"]* "\\""
        float ::= "-"? [0-9]+ "." [0-9]+
        ws ::= [ \\t\\n\\r]*
        """
        return LlamaGrammar.from_string(grammar_text)

    def analyze_synergy_batch(self, card_pairs: List[Tuple[str, str]],
                              batch_size: int = 100) -> List[Dict]:
        """Main entry point: batch analysis with cache & fallback."""
        results = []
        for i in range(0, len(card_pairs), batch_size):
            batch = card_pairs[i:i+batch_size]
            # Check cache first
            cached = [self.cache.get_synergy(a, b) for a, b in batch]
            valid_cached = [r for r in cached if r is not None]
            results.extend(valid_cached)

            # Determine uncached pairs
            cached_keys = {(r["card_a"], r["card_b"]) for r in valid_cached}
            uncached = [(a, b) for a, b in batch if (a, b) not in cached_keys]

            if uncached:
                try:
                    llm_results = self._llm_analyze(uncached)
                    # Cache and add to results
                    self.cache.cache_synergy_batch(llm_results)
                    results.extend(llm_results)
                except Exception as e:
                    print(f"LLM analysis failed: {e}. Falling back to heuristic.")
                    heuristic = self._heuristic_analyze(uncached)
                    self.cache.cache_synergy_batch(heuristic)
                    results.extend(heuristic)
        return results

    def _llm_analyze(self, pairs: List[Tuple[str, str]]) -> List[Dict]:
        """Generate synergy using LLM with constrained grammar."""
        prompt = self.prompt_builder.build_synergy_prompt(pairs, self.db)
        # Stream to non-blocking queue; we collect full response here.
        response = ""
        for token in self.model(prompt, grammar=self.grammar, stream=True):
            response += token["choices"][0]["text"]
            if len(response) > 4000:  # safety limit
                break

        # Parse and validate
        parsed = self._parse_json(response)
        if parsed is None:
            raise ValueError("LLM output could not be parsed as JSON")
        validated = validate_card_names(parsed, self.db)
        return validated

    def _parse_json(self, text: str) -> Optional[Dict]:
        """Extract JSON object from model output (handles trailing text)."""
        # Find the first '{' and last '}'
        start = text.find('{')
        end = text.rfind('}')
        if start == -1 or end == -1 or end <= start:
            return None
        json_str = text[start:end+1]
        try:
            return json.loads(json_str)
        except json.JSONDecodeError:
            return None

    def _heuristic_analyze(self, pairs: List[Tuple[str, str]]) -> List[Dict]:
        """Fallback: keyword/tribal based synergy scoring."""
        from ..engine.synergy_scorer import HeuristicScorer  # lazy import
        scorer = HeuristicScorer(self.db)
        return scorer.score_batch(pairs)
```

```python
            "card_b": card_b,
```

```python
class SynergyCache:
    def __init__(self, db_path: str = "cache/synergies.db", memory_size: int = 10000, ttl: int = 300):
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self.conn.executescript("""
            CREATE TABLE IF NOT EXISTS synergy_cache (
                card_a TEXT,
                card_b TEXT,
                score REAL,
                reason TEXT,
                cached_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                PRIMARY KEY (card_a, card_b)
            );
            PRAGMA journal_mode=WAL;
        """)
        self.lock = threading.Lock()
        self.memory_cache = OrderedDict()
        self.memory_size = memory_size
        self.ttl = ttl

    def get_synergy(self, card_a: str, card_b: str) -> Optional[Dict]:
        key = tuple(sorted([card_a, card_b]))
        # Memory check
        if key in self.memory_cache:
            entry = self.memory_cache[key]
            if time.time() - entry['time'] < self.ttl:
                self.memory_cache.move_to_end(key)
                return entry['data']
            else:
                del self.memory_cache[key]

        # Disk check
        with self.lock:
            cursor = self.conn.execute(
                "SELECT score, reason FROM synergy_cache WHERE card_a=? AND card_b=?",
                (card_a, card_b)
            )
            row = cursor.fetchone()
            if row:
                data = {"card_a": card_a, "card_b": card_b, "score": row[0], "reason": row[1]}
```

```python
from typing import List, Dict, Tuple
from ..engine.card_db import MagicCardDB

class HeuristicScorer:
    def __init__(self, db: MagicCardDB):
        self.db = db

    def score_batch(self, pairs: List[Tuple[str, str]]) -> List[Dict]:
        results = []
        for a, b in pairs:
            score = self._pair_score(a, b)
            results.append({
                "card_a": a,
                "card_b": b,
                "score": score,
                "reason": "Heuristic fallback"
            })
        return results

    def _pair_score(self, a: str, b: str) -> float:
        # Simplified heuristic: check creature types, mana cost, keywords
        card_a = self._get_card(a)
        card_b = self._get_card(b)
        if not card_a or not card_b:
            return 0.0

        score = 0.0
        # Shared creature type (tribal)
        types_a = set(card_a.get('type_line', '').split())
        types_b = set(card_b.get('type_line', '').split())
        if types_a & types_b:
            score += 0.3

        # Keyword overlap (e.g., both have "flying")
        keywords_a = set(card_a.get('keywords', '').split())
        keywords_b = set(card_b.get('keywords', '').split())
        if keywords_a & keywords_b:
            score += 0.2

        # Mana curve synergy: one low CMC, one high CMC (ramp)
        cmc_a = card_a.get('cmc', 0)
        cmc_b = card_b.get('cmc', 0)
        if abs(cmc_a - cmc_b) >= 3 and min(cmc_a, cmc_b) <= 2:
            score += 0.2

        # Color compatibility
        colors_a = set(card_a.get('colors', ''))
        colors_b = set(card_b.get('colors', ''))
        if colors_a and colors_b and colors_a == colors_b:
            score += 0.1

        return min(score, 1.0)

    def _get_card(self, name: str) -> Dict:
        cursor = self.db.conn.execute("SELECT name, type_line, cmc, colors, keywords FROM cards WHERE name=?", (name,))
        row = cursor.fetchone()
        if row:
            return {
                "name": row[0],
                "type_line": row[1],
                "cmc": row[2],
                "colors": row[3],
                "keywords": row[4]
            }
        return None
```

## Phase 3 Implementation: Offline-First LLM Integration

This phase focuses on the core LLM module – `llm/local_llm.py` – plus supporting files for prompt engineering, response parsing, synergy caching, and fallback heuristics.  
All code assumes the `engine/` modules (card database, deck rules) are already in place.

### 1. Critical Answers to Phase 3 Questions

**Q1: Managing GGUF model with llama-cpp-python (CPU/GPU, caching, streaming)**  
- Use `llama_cpp.Llama` with `n_ctx=4096` for phi‑3‑mini / `n_ctx=8192` for mistral.  
- Offload GPU layers via `n_gpu_layers` (e.g., 35 for an 8‑GB card).  
- Enable `use_mmap=True` for fast memory mapping.  
- Implement a thread‑safe **response queue** and yield tokens one‑by‑one to the UI (non‑blocking).  
- Cache LLM responses in an **LRU cache** with disk persistence (SQLite) – see `cache_manager.py`.  
- For streaming, use `model(prompt, stream=True)` and yield tokens; the UI thread consumes them.

**Q2: Prompt system with structured JSON (GBNF grammar)**  
- Define a **GBNF grammar** that forces the LLM to output exactly:  
  `{"synergies": [{"card_a": str, "card_b": str, "score": float, "reason": str}, ...]}`  
- The grammar rejects free text, prevents hallucinated card names by only allowing strings that match known names (passed in the prompt).  
- Include **anti‑synergy** detection by instructing the model to assign scores < 0.3 for conflicting cards (e.g., two different tribal lords, or cards that counter each other).  
- If a card name is not in the provided list, the grammar can still accept it; we then post‑process and replace with `UNKNOWN_CARD`.

**Q3: Batching, caching, and fallback**  
- Batch up to 100 card pairs per LLM call (to fit in context).  
- Check **SQLite cache** first (disk + in‑memory LRU).  
- If LLM fails (timeout, error, corrupted output), fall back to **keyword‑based heuristic scorer** that checks:  
  - Shared creature types / subtypes  
  - Mana cost synergy (e.g., low‑CMC spells + high‑CMC payoffs)  
  - Explicit combo keywords (“whenever you cast”, “when this enters”, etc.)  
- Heuristic scoring is fast and runs locally without the LLM.

**Q4: Anti‑hallucination techniques**  
- **Truncated card list**: In the prompt we include only the names of cards from the current deck plus a small context set (no full database).  
- **Explicit instruction**: “You ONLY analyze cards from this exact list. Do not invent any cards.”  
- **Post‑processing regex**: After parsing JSON, we verify every `card_a` / `card_b` against a set of known names (loaded from the DB). Non‑matching names are replaced with `UNKNOWN_CARD` and score set to 0.0.  
- **Grammar constraints**: The GBNF grammar limits characters; names with special characters are still allowed, but the model cannot inject extra text.

```python
import json
import queue
import threading
import time
from typing import List, Dict, Optional, Tuple
from llama_cpp import Llama
from llama_cpp.llama_grammar import LlamaGrammar
from .response_parser import validate_card_names
from .prompt_templates import SynergyPromptBuilder
from ..performance.cache_manager import SynergyCache
from ..engine.card_db import MagicCardDB  # for known card names

class OfflineMagicLLM:
    def __init__(self,
                 model_path: str = "models/phi-3-mini-4k-instruct.Q4_K_M.gguf",
                 db_path: str = "mtg_cards.db",
                 cache_path: str = "cache/synergies.db",
                 n_gpu_layers: int = 0,    # set >0 for GPU offloading
                 n_threads: int = 8,
                 n_ctx: int = 4096):
        self.db = MagicCardDB(db_path)
        self.cache = SynergyCache(cache_path)
        self.prompt_builder = SynergyPromptBuilder()
        self.model = Llama(
            model_path=model_path,
            n_ctx=n_ctx,
            n_threads=n_threads,
            n_gpu_layers=n_gpu_layers,
            verbose=False,
            use_mmap=True
        )
        self.grammar = self._load_grammar()
        self.response_queue = queue.Queue()
        self.lock = threading.Lock()

    def _load_grammar(self) -> LlamaGrammar:
        """GBNF grammar for structured synergy output."""
        grammar_text = """
        root ::= "{" ws "\\"synergies\\"" ws ":" ws "[" ws synergy_pair ("," ws synergy_pair)* ws "]" ws "}"
        synergy_pair ::= "{" ws "\\"card_a\\"" ws ":" ws string ws "," ws "\\"card_b\\"" ws ":" ws string ws "," ws "\\"score\\"" ws ":" ws float ws "," ws "\\"reason\\"" ws ":" ws string ws "}"
        string ::= "\\"" [^\\"]* "\\""
        float ::= "-"? [0-9]+ "." [0-9]+
        ws ::= [ \\t\\n\\r]*
        """
        return LlamaGrammar.from_string(grammar_text)

    def analyze_synergy_batch(self, card_pairs: List[Tuple[str, str]],
                              batch_size: int = 100) -> List[Dict]:
        """Main entry point: batch analysis with cache & fallback."""
        results = []
        for i in range(0, len(card_pairs), batch_size):
            batch = card_pairs[i:i+batch_size]
            # Check cache first
            cached = [self.cache.get_synergy(a, b) for a, b in batch]
            valid_cached = [r for r in cached if r is not None]
            results.extend(valid_cached)

            # Determine uncached pairs
            cached_keys = {(r["card_a"], r["card_b"]) for r in valid_cached}
            uncached = [(a, b) for a, b in batch if (a, b) not in cached_keys]

            if uncached:
                try:
                    llm_results = self._llm_analyze(uncached)
                    # Cache and add to results
                    self.cache.cache_synergy_batch(llm_results)
                    results.extend(llm_results)
                except Exception as e:
                    print(f"LLM analysis failed: {e}. Falling back to heuristic.")
                    heuristic = self._heuristic_analyze(uncached)
                    self.cache.cache_synergy_batch(heuristic)
                    results.extend(heuristic)
        return results

    def _llm_analyze(self, pairs: List[Tuple[str, str]]) -> List[Dict]:
        """Generate synergy using LLM with constrained grammar."""
        prompt = self.prompt_builder.build_synergy_prompt(pairs, self.db)
        # Stream to non-blocking queue; we collect full response here.
        response = ""
        for token in self.model(prompt, grammar=self.grammar, stream=True):
            response += token["choices"][0]["text"]
            if len(response) > 4000:  # safety limit
                break

        # Parse and validate
        parsed = self._parse_json(response)
        if parsed is None:
            raise ValueError("LLM output could not be parsed as JSON")
        validated = validate_card_names(parsed, self.db)
        return validated

    def _parse_json(self, text: str) -> Optional[Dict]:
        """Extract JSON object from model output (handles trailing text)."""
        # Find the first '{' and last '}'
        start = text.find('{')
        end = text.rfind('}')
        if start == -1 or end == -1 or end <= start:
            return None
        json_str = text[start:end+1]
        try:
            return json.loads(json_str)
        except json.JSONDecodeError:
            return None

    def _heuristic_analyze(self, pairs: List[Tuple[str, str]]) -> List[Dict]:
        """Fallback: keyword/tribal based synergy scoring."""
        from ..engine.synergy_scorer import HeuristicScorer  # lazy import
        scorer = HeuristicScorer(self.db)
        return scorer.score_batch(pairs)
```

```python
import json

class SynergyPromptBuilder:
    def build_synergy_prompt(self, pairs: List[Tuple[str, str]], db: MagicCardDB) -> str:
        """Build a prompt with only known card names to prevent hallucination."""
        all_names = set()
        for a, b in pairs:
            all_names.add(a)
            all_names.add(b)

        # Get valid cards from database (truncated list)
        valid_cards = self._get_known_cards(list(all_names), db)

        # Build prompt
        sys_msg = (
            "You are a Magic: The Gathering deck synergy analyzer. "
            "You ONLY analyze cards from this exact list (do not invent any cards):\n\n"
            f"Known cards: {json.dumps(valid_cards)}\n\n"
            "Analyze these card pairs for synergy (0.0 = no synergy, 1.0 = perfect synergy). Consider:\n"
            "- Combo potential (infinite combos, two-card combos)\n"
            "- Tribal synergy (shares creature type)\n"
            "- Mana curve interaction\n"
            "- Color fixing needs\n"
            "- Theme alignment (graveyard, artifacts, spellslinger, etc.)\n"
            "Assign a score between 0.0 and 1.0. Provide a brief reason.\n\n"
            "Pairs to analyze:\n"
        )

        pairs_text = json.dumps(pairs)
        user_msg = (
            f"{pairs_text}\n\n"
            "Respond ONLY with valid JSON matching this exact schema (no other text):\n"
            '{"synergies": [{"card_a": string, "card_b": string, "score": float, "reason": string}, ...]}\n\n'
            "IMPORTANT: If a card name is NOT in the provided list, use 'UNKNOWN_CARD' instead and assign score 0.0."
        )

        # Using llama.cpp chat format (e.g., for Mistral/Phi)
        return f"<s>[INST] {sys_msg}\n{user_msg} [/INST]"

    def _get_known_cards(self, names: List[str], db: MagicCardDB) -> List[str]:
        """Filter names to those that actually exist in the database."""
        # Use SQL to check existence
        placeholders = ','.join('?' for _ in names)
        cursor = db.conn.execute(f"SELECT name FROM cards WHERE name IN ({placeholders})", names)
        known = {row[0] for row in cursor.fetchall()}
        return [n for n in names if n in known]  # preserve order
```

```python
def validate_card_names(parsed: Dict, db: MagicCardDB) -> List[Dict]:
    """Post‑process: replace hallucinated names with UNKNOWN_CARD and clamp scores."""
    # Build set of real card names from DB (can be cached)
    real_names = _get_all_card_names(db)

    validated = []
    for item in parsed.get("synergies", []):
        card_a = item.get("card_a", "UNKNOWN_CARD")
        card_b = item.get("card_b", "UNKNOWN_CARD")
        score = item.get("score", 0.0)
        reason = item.get("reason", "")

        # Replace if not real
        if card_a not in real_names:
            card_a = "UNKNOWN_CARD"
            score = 0.0
        if card_b not in real_names:
            card_b = "UNKNOWN_CARD"
            score = 0.0

        # Clamp score
        score = max(0.0, min(1.0, float(score)))

        validated.append({
            "card_a": card_a,
            "card_b": card_b,
            "score": score,
            "reason": reason
        })
    return validated

        _get_all_card_names.cache = {row[0] for row in cursor.fetchall()}
```

```python
import sqlite3
import threading
import time
from collections import OrderedDict
from typing import Optional, Dict, List

class SynergyCache:
    def __init__(self, db_path: str = "cache/synergies.db", memory_size: int = 10000, ttl: int = 300):
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self.conn.executescript("""
            CREATE TABLE IF NOT EXISTS synergy_cache (
                card_a TEXT,
                card_b TEXT,
                score REAL,
                reason TEXT,
                cached_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                PRIMARY KEY (card_a, card_b)
            );
            PRAGMA journal_mode=WAL;
        """)
        self.lock = threading.Lock()
        self.memory_cache = OrderedDict()
        self.memory_size = memory_size
        self.ttl = ttl

    def get_synergy(self, card_a: str, card_b: str) -> Optional[Dict]:
        key = tuple(sorted([card_a, card_b]))
        # Memory check
        if key in self.memory_cache:
            entry = self.memory_cache[key]
            if time.time() - entry['time'] < self.ttl:
                self.memory_cache.move_to_end(key)
                return entry['data']
            else:
                del self.memory_cache[key]

        # Disk check
        with self.lock:
            cursor = self.conn.execute(
                "SELECT score, reason FROM synergy_cache WHERE card_a=? AND card_b=?",
                (card_a, card_b)
            )
            row = cursor.fetchone()
            if row:
                data = {"card_a": card_a, "card_b": card_b, "score": row[0], "reason": row[1]}
                self._add_to_memory(key, data)
                return data
        return None

    def cache_synergy_batch(self, results: List[Dict]):
        with self.lock:
            self.conn.executemany(
                "INSERT OR REPLACE INTO synergy_cache (card_a, card_b, score, reason) VALUES (?, ?, ?, ?)",
                [(r['card_a'], r['card_b'], r['score'], r['reason']) for r in results]
            )
            self.conn.commit()
        for r in results:
            key = tuple(sorted([r['card_a'], r['card_b']]))
            self._add_to_memory(key, r)

    def _add_to_memory(self, key: tuple, data: Dict):
        if len(self.memory_cache) >= self.memory_size:
            self.memory_cache.popitem(last=False)
        self.memory_cache[key] = {'data': data, 'time': time.time()}
```

```python
from typing import List, Dict, Tuple
from ..engine.card_db import MagicCardDB

class HeuristicScorer:
    def __init__(self, db: MagicCardDB):
        self.db = db

    def score_batch(self, pairs: List[Tuple[str, str]]) -> List[Dict]:
        results = []
        for a, b in pairs:
            score = self._pair_score(a, b)
            results.append({
                "card_a": a,
                "card_b": b,
                "score": score,
                "reason": "Heuristic fallback"
            })
        return results

    def _pair_score(self, a: str, b: str) -> float:
        # Simplified heuristic: check creature types, mana cost, keywords
        card_a = self._get_card(a)
        card_b = self._get_card(b)
        if not card_a or not card_b:
            return 0.0

        score = 0.0
        # Shared creature type (tribal)
        types_a = set(card_a.get('type_line', '').split())
        types_b = set(card_b.get('type_line', '').split())
        if types_a & types_b:
            score += 0.3

        # Keyword overlap (e.g., both have "flying")
        keywords_a = set(card_a.get('keywords', '').split())
        keywords_b = set(card_b.get('keywords', '').split())
        if keywords_a & keywords_b:
            score += 0.2

        # Mana curve synergy: one low CMC, one high CMC (ramp)
        cmc_a = card_a.get('cmc', 0)
        cmc_b = card_b.get('cmc', 0)
        if abs(cmc_a - cmc_b) >= 3 and min(cmc_a, cmc_b) <= 2:
            score += 0.2

        # Color compatibility
        colors_a = set(card_a.get('colors', ''))
        colors_b = set(card_b.get('colors', ''))
        if colors_a and colors_b and colors_a == colors_b:
            score += 0.1

        return min(score, 1.0)

    def _get_card(self, name: str) -> Dict:
        cursor = self.db.conn.execute("SELECT name, type_line, cmc, colors, keywords FROM cards WHERE name=?", (name,))
        row = cursor.fetchone()
        if row:
            return {
                "name": row[0],
                "type_line": row[1],
                "cmc": row[2],
                "colors": row[3],
                "keywords": row[4]
            }
        return None
```

```python
            if uncached:
                try:
                    llm_results = self._llm_analyze(uncached)
                    self.cache.cache_synergy_batch(llm_results)
                    results.extend(llm_results)
                except Exception as e:
                    print(f"LLM failed: {e}. Using heuristic.")
                    heuristic = self._heuristic_analyze(uncached)
                    self.cache.cache_synergy_batch(heuristic)
                    results.extend(heuristic)
        return results

    def _llm_analyze(self, pairs: List[Tuple[str,str]]) -> List[Dict]:
        prompt = self.prompt_builder.build_synergy_prompt(pairs, self.db)
        response = ""
        for token in self.model(prompt, grammar=self.grammar, stream=True):
            response += token["choices"][0]["text"]
            if len(response) > 4000:
                break
        parsed = self._parse_json(response)
        if parsed is None:
            raise ValueError("Invalid JSON output")
        return validate_card_names(parsed, self.db)

    def _parse_json(self, text: str) -> Optional[Dict]:
        start = text.find('{')
        end = text.rfind('}')
        if start == -1 or end == -1 or end <= start:
            return None
        try:
            return json.loads(text[start:end+1])
        except json.JSONDecodeError:
            return None

    def _heuristic_analyze(self, pairs: List[Tuple[str,str]]) -> List[Dict]:
        from ..engine.synergy_scorer import HeuristicScorer
        scorer = HeuristicScorer(self.db)
        return scorer.score_batch(pairs)
```

```python
from typing import List, Dict, Tuple
from ..engine.card_db import MagicCardDB

class HeuristicScorer:
    def __init__(self, db: MagicCardDB):
        self.db = db
    def score_batch(self, pairs: List[Tuple[str,str]]) -> List[Dict]:
        return [{"card_a":a,"card_b":b,"score":self._pair_score(a,b),"reason":"heuristic"}
                for a,b in pairs]
    def _pair_score(self, a: str, b: str) -> float:
        ca = self._get_card(a); cb = self._get_card(b)
        if not ca or not cb: return 0.0
        s = 0.0
        types_a = set(ca.get('type_line','').split()); types_b = set(cb.get('type_line','').split())
        if types_a & types_b: s += 0.3
        kw_a = set(ca.get('keywords','').split()); kw_b = set(cb.get('keywords','').split())
        if kw_a & kw_b: s += 0.2
        cmc_a = ca.get('cmc',0); cmc_b = cb.get('cmc',0)
        if abs(cmc_a - cmc_b) >= 3 and min(cmc_a, cmc_b) <= 2: s += 0.2
        col_a = set(ca.get('colors','')); col_b = set(cb.get('colors',''))
        if col_a and col_b and col_a == col_b: s += 0.1
        return min(s, 1.0)
    def _get_card(self, name: str) -> Dict:
        cur = self.db.conn.execute("SELECT name,type_line,cmc,colors,keywords FROM cards WHERE name=?", (name,))
        row = cur.fetchone()
        if row:
            return {"name":row[0],"type_line":row[1],"cmc":row[2],"colors":row[3],"keywords":row[4]}
        return None
```

--- performance/cache_manager.py ---
```python
import sqlite3, threading, time
from collections import OrderedDict
from typing import Optional, Dict, List

class SynergyCache:
    def __init__(self, db_path="cache/synergies.db", memory_size=10000, ttl=300):
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self.conn.executescript("""
            CREATE TABLE IF NOT EXISTS synergy_cache (
                card_a TEXT, card_b TEXT, score REAL, reason TEXT,
                cached_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                PRIMARY KEY (card_a, card_b));
            PRAGMA journal_mode=WAL;""")
        self.lock = threading.Lock()
        self.memory_cache = OrderedDict()
        self.memory_size = memory_size
        self.ttl = ttl

    def get_synergy(self, card_a: str, card_b: str) -> Optional[Dict]:
        key = tuple(sorted([card_a, card_b]))
        if key in self.memory_cache:
            entry = self.memory_cache[key]
            if time.time() - entry['time'] < self.ttl:
                self.memory_cache.move_to_end(key)
                return entry['data']
            else:
                del self.memory_cache[key]
        with self.lock:
            cur = self.conn.execute("SELECT score, reason FROM synergy_cache WHERE card_a=? AND card_b=?", (card_a, card_b))
            row = cur.fetchone()
            if row:
                data = {"card_a":card_a,"card_b":card_b,"score":row[0],"reason":row[1]}
                self._add_to_memory(key, data)
                return data
        return None

    def cache_synergy_batch(self, results: List[Dict]):
        with self.lock:
            self.conn.executemany("INSERT OR REPLACE INTO synergy_cache (card_a,card_b,score,reason) VALUES (?,?,?,?)",
                                  [(r['card_a'],r['card_b'],r['score'],r['reason']) for r in results])
            self.conn.commit()
        for r in results:
            key = tuple(sorted([r['card_a'], r['card_b']]))
            self._add_to_memory(key, r)

    def _add_to_memory(self, key: tuple, data: Dict):
        if len(self.memory_cache) >= self.memory_size:
            self.memory_cache.popitem(last=False)
        self.memory_cache[key] = {'data': data, 'time': time.time()}
```
